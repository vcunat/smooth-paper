% ensures text mode, the opposite of \ensuremath
\def\nomath#1{\ifmmode\text{#1}\else#1\fi}


\def\interval#1#2{\left\langle#1,#2\right\rangle}
\def\intervalOC#1#2{\left(#1,#2\right\rangle}
\def\intervalOO#1#2{\left(#1,#2\right)}

\def\eqspaces{\;=\;}
\def\lespaces{\;\le\;}

\def\Prob#1{\Pr\!\left[#1\right]}
\def\ProbParam#1#2{\Pr_{#1}\!\left[#2\right]}
\def\Exp#1{\mathrm{E}\left[#1\right]}

\def\OO{\mathcal O}
\def\oo{\ensuremath o}

\def\diff#1{\mathrm{d}#1}
\def\euler{\mathrm{e}}
\def\natural{\mathbb{N}}

% emphasizing the defined term in text mode
\def\emphDef#1{{\bf#1}}
% emphasize in the math mode (\emphDef doesn't work there)
\def\emphDefMath#1{\mathbf{#1}}

\def\setZeroTo#1{\{0,\dotsc,{#1}\}}


%% various styles for algorithmic texts
	% structure fields and methods, functions
	\def\aField#1{\nomath{\textsc{#1}}}
	\def\aMethod#1{\nomath{\textsf{#1}}}
	\def\aFunc{\nomath}

	% assignment, variable definition with initialization
	\def\aAssign#1{\STATE #1 $\leftarrow$ }
	\def\aVar#1{\STATE var $ #1 \leftarrow $ }

	% some keywords
	\def\aKeyword#1{\textmd{#1}}
	\def\aNULL{\aKeyword{null}}
	\def\aFALSE{\aKeyword{false}}
	\def\aTRUE{\aKeyword{true}}
	\def\aOR{\;\aKeyword{or}\;}
	\def\aAND{\;\aKeyword{and}\;}
	\def\aNOT{\aKeyword{not}}
	\def\aTHIS{\aKeyword{this}}

	% comments for a block of code, for a line and a commented if
	\def\formatComment#1{{\footnotesize#1}}
	\def\aCommBlock#1{\medskip\STATE\COMMENT{\formatComment{\normalsize#1}}\smallskip}
	\def\aCommLine#1{\hspace{1em minus 0.5em}\{\formatComment{#1}\}}
